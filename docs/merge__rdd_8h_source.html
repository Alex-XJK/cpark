154 constexpr MergeRdd(const R& prev) : Base{prev, false} {
155 static_assert(concepts::Rdd<MergeRdd<R>>,
156 "Instance of MergeRdd does not satisfy Rdd concept.");
157 // Prepare nested splits vector
158 using OriginalIterator = std::ranges::iterator_t<std::ranges::range_value_t<R>>;
159 std::vector<OriginalIterator> all_prev_splits_begins;
160 std::vector<OriginalIterator> all_prev_splits_ends;
161 for (const concepts::Split auto& prev_split : prev) {
162 all_prev_splits_begins.emplace_back(std::ranges::begin(prev_split));
163 all_prev_splits_ends.emplace_back(std::ranges::end(prev_split));
164 }
165
166 // Create the single splits_ element
167 splits_.emplace_back(
168 std::ranges::subrange{
169 Iterator{all_prev_splits_begins, all_prev_splits_ends, false},
170 Iterator{all_prev_splits_begins, all_prev_splits_ends, true},
171 },
172 prev.front());
173 for (const concepts::Split auto& prev_split : prev)
174 splits_.back().addDependency(prev_split);
175 }
